import pandas as pd
from sklearn.linear_model import LinearRegression

# Dataset
data = {
    'Hours': [1, 2, 3, 4, 5, 6],
    'Battery_Usage': [15, 30, 45, 60, 75, 90]
}

df = pd.DataFrame(data)

X = df[['Hours']]
y = df['Battery_Usage']

# Create model
model = LinearRegression()

# Train model
model.fit(X, y)

# Predict battery usage for 7 hours
prediction = model.predict([[7]])

print("Predicted Battery Usage:", prediction[0], "%")
